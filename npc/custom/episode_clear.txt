function	script	F_CheckQuestState	{
	.@id = getarg(0);
	.@state = checkquest(.@id, PLAYTIME);
	if (.@state == 2) return "^4d4dffCompleted^000000";
	return "Incomplete";
}

// Returns true when the player finished this episode through official story scripts
// (variables / story quest IDs), without Balheele chapter marker quests.
function	script	F_EpOrganicComplete	{
	.@ep = getarg(0);
	.@part = getarg(1, 0);

	switch (.@ep) {
	case 13:
		return (ep13_ryu > 99 || ep13_start > 99 || ep13_mdrama > 26);
	case 14:
		return (ep14_3_newerabs >= 24);
	case 15:
		return (ep15_2_bslast >= 9 || ep15_1_atnad >= 33);
	case 16:
		return (terra_gloria_main >= 25);
	case 17:
		if (.@part == 2)
			return (isbegin_quest(12497) == 2 || ep17_2_main >= 36);
		return (isbegin_quest(16360) == 2 || ep17_2_main >= 1);
	case 18:
		return (ep18_main >= 57);
	case 19:
		return (ep19_main >= 100);
	case 20:
		return (ep20_main >= 100 || checkquest(20008, PLAYTIME) == 2);
	}
	return 0;
}

// Prior-episode prerequisite: chapter marker quest OR organic story completion.
function	script	F_EpPrereqMet	{
	.@marker_id = getarg(0);

	if (!.@marker_id)
		return 1;
	if (checkquest(.@marker_id, PLAYTIME) == 2)
		return 1;

	switch (.@marker_id) {
	case 13223: return callfunc("F_EpOrganicComplete", 13, 0);
	case 14002: return callfunc("F_EpOrganicComplete", 14, 0);
	case 15013: return callfunc("F_EpOrganicComplete", 15, 0);
	case 18219: return callfunc("F_EpOrganicComplete", 16, 0);
	case 17046: return callfunc("F_EpOrganicComplete", 17, 1);
	case 17108: return callfunc("F_EpOrganicComplete", 17, 2);
	case 18164: return callfunc("F_EpOrganicComplete", 18, 0);
	case 19210: return callfunc("F_EpOrganicComplete", 19, 0);
	}
	return 0;
}

function	script	F_EpEpisodeComplete	{
	.@ep = getarg(0);
	.@part = getarg(1, 0);
	.@marker_id = getarg(2, 0);

	if (.@marker_id && checkquest(.@marker_id, PLAYTIME) == 2)
		return 1;
	return callfunc("F_EpOrganicComplete", .@ep, .@part);
}

function	script	F_EpForceComplete	{
	.@id = getarg(0);

	if (checkquest(.@id) != -1)
		erasequest .@id;

	setquest .@id;

	.@state = checkquest(.@id);
	if (.@state >= 0 && .@state < 2)
		completequest .@id;
	return;
}

			}

			for (.@t = 0; .@t < getarraysize(.@tickets); .@t++) {
				if (countitem(.@tickets[.@t]) > 0) {
					.@ticket_used = .@tickets[.@t];
					break;
				}
			}
			if (!.@ticket_used) {
				mes "Hmm...";
				mes "Don't you have any of these tickets?";
				.@icon$ = "";
				for (.@t = 0; .@t < getarraysize(.@tickets); .@t++)
					.@icon$ += " ^i[" + .@tickets[.@t] + "]";
				mes .@icon$;
				mes "I consume one ticket, but I can exchange it for you.";
				close;
			}

			for (.@i = .@start; .@i <= .@end; .@i++)
				callfunc("F_EpForceComplete", .@ids[.@i]);
			callfunc("F_EpClearApply", 17, .@part);
			delitem .@ticket_used, 1;
			callfunc("F_EpClearConfirm", 17, .@part, .@ticket_used);
			end;
	}
	end;
}
